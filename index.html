////////////////////////////// // ATEŞOĞLU MARKET PRO // STOK YÖNETİM MODÜLÜ ////////////////////////////// const STORAGE_KEY = "atesogluStocks"; // 📦 STOKLARI ÇEK function getStocks() { return JSON.parse(localStorage.getItem(STORAGE_KEY)) || []; } // 💾 STOKLARI KAYDET function saveStocks(stocks) { localStorage.setItem(STORAGE_KEY, JSON.stringify(stocks)); } // ➕ STOK EKLE (varsa artır, yoksa ekle) function addStock(name, qty) { let stocks = getStocks(); let existing = stocks.find(item => item.name === name); if (existing) { existing.quantity += Number(qty); } else { stocks.push({ name: name, quantity: Number(qty) }); } saveStocks(stocks); renderStocks(); } // 🗑️ TEK STOK SİL function deleteStock(index) { let stocks = getStocks(); stocks.splice(index, 1); saveStocks(stocks); renderStocks(); } // 🧨 TÜM STOKLARI SIFIRLA function deleteAllStocks() { if (!confirm("TÜM ATEŞOĞLU MARKET STOKLARI SİLİNECEK. Emin misin?")) return; localStorage.removeItem(STORAGE_KEY); renderStocks(); } // 📊 STOK LİSTELE function renderStocks() { let stocks = getStocks(); let table = document.getElementById("stockTable"); if (!table) return; table.innerHTML = ""; stocks.forEach((item, i) => { table.innerHTML += ` ${item.name} ${item.quantity} `; }); } // 🧾 INPUT'TAN STOK EKLE (UI bağlantısı) function addStockFromUI() { let name = document.getElementById("productName").value; let qty = document.getElementById("productQty").value; if (!name || !qty) { alert("Ürün adı ve miktar gir"); return; } addStock(name, qty); document.getElementById("productName").value = ""; document.getElementById("productQty").value = ""; } // 🚀 BAŞLAT renderStocks();
